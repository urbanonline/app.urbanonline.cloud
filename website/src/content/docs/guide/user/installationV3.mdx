<script>
{`
  (function () {
    const TOTAL = 7; // Schritte 1–6 + Abschluss
    let current = 1;

    function stepId(n) {
      return n <= 6 ? 'step-' + n : 'step-finish';
    }

    function show(n) {
      for (let i = 1; i <= TOTAL; i++) {
        const el = document.getElementById(stepId(i));
        if (el) el.hidden = i !== n;
      }

      // Fortschrittsanzeige
      const prog = document.getElementById('ufm-progress');
      if (prog) prog.textContent = n <= 6
        ? 'Schritt ' + n + ' von 6'
        : 'Einrichtung abgeschlossen';

      // Navigations-Buttons
      const prev = document.getElementById('ufm-btn-prev');
      const next = document.getElementById('ufm-btn-next');
      if (prev) prev.disabled = n === 1;
      if (next) {
        next.disabled = n === TOTAL;
        next.textContent = n === 6 ? 'Abschließen →' : 'Weiter →';
      }

      // Aktiven Step-Button hervorheben
      document.querySelectorAll('.ufm-step-btn').forEach(btn => {
        const s = parseInt(btn.dataset.step, 10);
        btn.classList.toggle('ufm-active', s === n);
        btn.setAttribute('aria-current', s === n ? 'step' : 'false');
      });

      current = n;
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    // Step-Buttons in der Nav
    document.querySelectorAll('.ufm-step-btn').forEach(btn => {
      btn.addEventListener('click', () => show(parseInt(btn.dataset.step, 10)));
    });

    // Globale Funktion für onclick-Attribute
    window.ufmNav = function (dir) {
      const next = current + dir;
      if (next >= 1 && next <= TOTAL) show(next);
    };

    show(1);
  })();
`}
</script>
